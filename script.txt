local url = "https://raw.githubusercontent.com/E5alR9/E5s/refs/heads/main/main"
local success, result = pcall(function()
    return game:HttpGet(url:gsub(" ", "%%20"))
end)

if success then
    local func, err = loadstring(result)
    if func then
        func()
    else
        warn("腳本語法錯誤: " .. err)
    end
else
    warn("無法連線至 GitHub，請檢查網址或網路狀態。")
end
