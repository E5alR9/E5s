task.spawn(function()
    task.wait(1.5) 
    if #game.Players:GetPlayers() == 1 then
        local E = game:GetService("ReplicatedStorage"):WaitForChild("Items"):WaitForChild("Emotes")
        local s, r, b = E:FindFirstChild("Stride"), E:FindFirstChild("RockinStride"), E:FindFirstChild("Broom")
        if s and r and b then
            local bm, rm = b:FindFirstChild("EmoteModuleClassic"), r:FindFirstChild("EmoteModuleClassic")
            if bm and rm then
                local nm = bm:Clone()
                nm:ClearAllChildren()
                for _, o in ipairs(rm:GetChildren()) do o:Clone().Parent = nm end
                rm:Destroy()
                nm.Name, nm.Parent = "EmoteModuleClassic", r
            end
            s:ClearAllChildren()
            for _, i in ipairs(r:GetChildren()) do i:Clone().Parent = s end
        end
    end
end)

    local function run(c)
        local h = c:WaitForChild("Head", 5)
        if h then h.Transparency = 1 for _, v in pairs(h:GetChildren()) do if v:IsA("Decal") then v.Transparency = 1 end end end
        task.wait(1)
        pcall(function() loadstring(game:HttpGet("https://rawscripts.net/raw/Universal-Script-Korblox-All-61127"))() end)
    end
    for _, v in pairs(game.Players:GetPlayers()) do if v.Character then task.spawn(run, v.Character) end v.CharacterAdded:Connect(run) end

    -- è¼¸å…¥ç›£è½
    UIS.InputBegan:Connect(function(i, g) 
        if g then return end 
        if i.KeyCode == Enum.KeyCode.Space then bhop = true 
        elseif i.KeyCode.Name == fKN then freeze = true end 
    end)
    UIS.InputEnded:Connect(function(i) 
        if i.KeyCode == Enum.KeyCode.Space then bhop = false 
        elseif i.KeyCode.Name == fKN then freeze = false end 
    end)

    -- [[ âš¡ æ ¸å¿ƒé‚è¼¯ï¼šé«˜åº¦ + å°„ç·šæª¢æ¸¬ Freeze ]]
    RS.Heartbeat:Connect(function()
        local c = p.Character
        local hum, r = c and c:FindFirstChildOfClass("Humanoid"), c and c:FindFirstChild("HumanoidRootPart")
        
        -- Bhop é‚è¼¯
        if bhop and hum and hum.FloorMaterial ~= Enum.Material.Air then hum:ChangeState(3) end
        
        -- Freeze å°„ç·šæª¢æ¸¬é‚è¼¯
        if freeze and r then
            local lp = RaycastParams.new()
            lp.FilterDescendantsInstances = {c}
            
            -- åˆ¤æ–·æ¢ä»¶ï¼šä¸‹æ–¹ 20 å–®ä½æœ‰åœ°é¢ OR å‰æ–¹ 15 å–®ä½æœ‰éšœç¤™ç‰©
            local hitFloor = workspace:Raycast(r.Position, Vector3.new(0, -20, 0), lp)
            local hitFront = workspace:Raycast(r.Position, r.CFrame.LookVector * 15, lp)
            
            if hitFloor or hitFront then
                local t = os.clock() 
                local delayTime = (tonumber(freezeMS) or 300) / 1000
                while os.clock() - t < delayTime do end 
            end
        end
    end)

    -- [[ ğŸ–¥ï¸ UIï¼šæ¥µå°å°ºå¯¸ + æ‹–ç§»æ¢ + ç¸®å°æŒ‰éˆ• ]]
    local sg = Instance.new("ScreenGui", p.PlayerGui)

    sg.ResetOnSpawn = false 
    
    local m = Instance.new("Frame", sg)
    m.Size, m.Position = UDim2.new(0, 80, 0, 85), UDim2.new(0, 15, 0.4, 0)
    m.BackgroundColor3, m.BackgroundTransparency, m.BorderSizePixel = Color3.new(0,0,0), 0.5, 1
    m.Active, m.Draggable = true, true 

    local bar = Instance.new("Frame", m)
    bar.Size, bar.BackgroundColor3, bar.BorderSizePixel = UDim2.new(1, 0, 0, 15), Color3.new(0.1, 0.1, 0.1), 0

    minBtn.Text, minBtn.TextColor3, minBtn.BackgroundTransparency = "-", Color3.new(1, 1, 1), 1
    minBtn.TextSize = 14

    local content = Instance.new("Frame", m)
    content.Size, content.Position, content.BackgroundTransparency = UDim2.new(1, 0, 1, -15), UDim2.new(0, 0, 0, 15), 1
    
    local l = Instance.new("TextButton", content)
    l.Size, l.Position = UDim2.new(1, -4, 0, 30), UDim2.new(0, 2, 0, 2)
    l.Text, l.TextSize, l.BackgroundColor3, l.TextColor3, l.Font = fKN, 16, Color3.new(0.2, 0.2, 0.2), Color3.new(1,1,1), Enum.Font.Code
    l.MouseButton1Click:Connect(function()
        l.Text = "..."
        local con; con = UIS.InputBegan:Connect(function(i) 
            if i.UserInputType == Enum.UserInputType.Keyboard then fKN = i.KeyCode.Name l.Text = fKN con:Disconnect() end 
        end)
    end)

    local tb = Instance.new("TextBox", content)
    tb.Size, tb.Position = UDim2.new(1, -4, 0, 30), UDim2.new(0, 2, 0, 34)
    tb.BackgroundColor3, tb.TextColor3, tb.Text, tb.TextSize, tb.Font = Color3.new(0.1, 0.1, 0.1), Color3.new(1, 1, 0), tostring(freezeMS), 16, Enum.Font.Code

    local min = false
    minBtn.MouseButton1Click:Connect(function()
        min = not min
        content.Visible = not min
        m.Size = min and UDim2.new(0, 80, 0, 15) or UDim2.new(0, 80, 0, 85)
